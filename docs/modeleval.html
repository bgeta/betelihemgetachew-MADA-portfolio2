#creating a receipe

flu_rec <-
  recipe(Nausea ~ ., data=train_data)

#create the model

lr_mod<-
  logistic_reg()%>%
  set_engine("glm")

#combine receipe and model

flu_wflow <-
  workflow() %>%
  add_model(lr_mod)%>%
  add_recipe(flu_rec)
